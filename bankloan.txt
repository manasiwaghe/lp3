pragma solidity ^0.8.0;

contract BankAccount {
    address public owner;

    // Mapping to store balance and loan details for each customer
    mapping(address => uint256) private balances;
    mapping(address => uint256) private loans;

    // Interest rate for loans (e.g., 5%)
    uint256 public constant interestRate = 5;

    // Events for transaction tracking
    event Deposit(address indexed customer, uint256 amount);
    event Withdraw(address indexed customer, uint256 amount);
    event LoanRequested(address indexed customer, uint256 amount);
    event LoanRepaid(address indexed customer, uint256 amount);
    event BankWithdrawal(address indexed owner, uint256 amount);

    // Constructor to set the contract owner
    constructor() {
        owner = msg.sender;
    }

    // Modifier to restrict actions to the contract owner
    modifier onlyOwner() {
        require(msg.sender == owner, "Only the owner can perform this action");
        _;
    }

    // Deposit Ether into your personal account
    function deposit() public payable {
        require(msg.value > 0, "Deposit amount must be greater than 0");
        balances[msg.sender] += msg.value;
        emit Deposit(msg.sender, msg.value);
    }

    // Withdraw Ether from your personal account
    function withdraw(uint256 amount) public {
        require(amount > 0, "Withdrawal amount must be greater than 0");
        require(balances[msg.sender] >= amount, "Insufficient balance");

        balances[msg.sender] -= amount;
        payable(msg.sender).transfer(amount);

        emit Withdraw(msg.sender, amount);
    }

    // Request a loan from the bank (credited to user's balance)
    function requestLoan(uint256 amount) public {
        require(amount > 0, "Loan amount must be greater than 0");
        require(address(this).balance >= amount, "Bank has insufficient funds");

        uint256 interest = (amount * interestRate) / 100;
        uint256 totalRepayable = amount + interest;

        loans[msg.sender] += totalRepayable;
        balances[msg.sender] += amount;

        emit LoanRequested(msg.sender, amount);
    }

    // Repay loan
    function repayLoan() public payable {
        require(loans[msg.sender] > 0, "No active loan to repay");
        require(msg.value > 0, "Repayment amount must be greater than 0");

        if (msg.value >= loans[msg.sender]) {
            uint256 extra = msg.value - loans[msg.sender];
            loans[msg.sender] = 0;
            balances[msg.sender] += extra; // Add overpayment to userâ€™s balance
        } else {
            loans[msg.sender] -= msg.value;
        }

        emit LoanRepaid(msg.sender, msg.value);
    }

    // Check your individual account balance
    function getBalance() public view returns (uint256) {
        return balances[msg.sender];
    }

    // Check your outstanding loan amount
    function getLoanBalance() public view returns (uint256) {
        return loans[msg.sender];
    }

    // Owner can view the total Ether held in the contract
    function getBankBalance() public view returns (uint256) {
        return address(this).balance;
    }

    // Owner-only function to withdraw funds from the bank (not user balances)
    function ownerWithdraw(uint256 amount) public onlyOwner {
        require(amount > 0, "Withdrawal amount must be greater than 0");
        require(address(this).balance >= amount, "Insufficient contract funds");

        payable(owner).transfer(amount);
        emit BankWithdrawal(owner, amount);
    }
}
